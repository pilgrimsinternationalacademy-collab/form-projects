<div class="container">

  <h2>세부 학습 코스 확정</h2>

  <p class="desc">등록이 완료된 자녀의 세부 교재 및 학습 폴더를 배정합니다.</p>

  

  <div id="loginArea">

    <label class="block-label">보호자 성함</label>

    <input type="text" id="gName" placeholder="예: 홍길동">

    

    <label class="block-label">보호자 연락처</label>

    <input type="tel" id="gContact" placeholder="예: 01012345678">

    

    <button class="btn btn-secondary" id="loginBtn" onclick="login()">학생 정보 불러오기</button>

  </div>


  <div id="courseArea">

    <div id="studentContainer"></div>

    <button class="btn btn-primary" id="submitBtn" onclick="submitCourses()">선택 코스 최종 제출하기</button>

  </div>

<script>

  // 💡 [필수] 방금 새로 배포한 SCRIPT_URL을 넣어주세요!

  const SCRIPT_URL = "여기에_원장님의_배포_URL을_넣으세요";

  

  let currentStudent = null;

  let loadedCourses = [];


  // [1단계] 보호자 정보로 학생 조회

  async function login() {

    const name = document.getElementById('gName').value.trim();

    const contact = document.getElementById('gContact').value.trim();

    if(!name || !contact) return alert("보호자 성함과 연락처를 입력해주세요.");


    const btn = document.getElementById('loginBtn');

    btn.disabled = true; btn.innerText = "조회 중...";


    try {

      const resp = await fetch(SCRIPT_URL, {

        method: 'POST',

        headers: { 'Content-Type': 'text/plain;charset=utf-8' },

        body: JSON.stringify({ action: 'lookup', guardianName: name, guardianContact: contact })

      });

      const data = await resp.json();

      

      if(data.result === 'not_found') return alert("일치하는 학생 정보를 찾을 수 없습니다.");


      // '등록 완료' 상태인 자녀만 필터링 (일단 첫 번째 자녀를 기준으로 세팅)

      const enrolledChildren = data.children.filter(c => c.status === '🔵 등록 완료');

      if(enrolledChildren.length === 0) return alert("아직 '등록 완료' 처리된 자녀가 없습니다. 학원으로 문의해주세요.");


      currentStudent = enrolledChildren[0]; // 다자녀일 경우 확장을 위한 베이스

      fetchCourseMap(currentStudent);


    } catch(e) { alert("통신 오류가 발생했습니다."); }

    finally { btn.disabled = false; btn.innerText = "학생 정보 불러오기"; }

  }


  // [2단계] 자녀의 학년과 프로그램에 맞는 Custom Programs 맵핑 가져오기

  async function fetchCourseMap(student) {

    document.getElementById('loginArea').style.display = 'none';

    document.getElementById('courseArea').style.display = 'block';

    

    const container = document.getElementById('studentContainer');

    container.innerHTML = `<p style="text-align:center;">맵핑된 코스를 불러오는 중입니다...</p>`;


    try {

      const resp = await fetch(SCRIPT_URL, {

        method: 'POST',

        headers: { 'Content-Type': 'text/plain;charset=utf-8' },

        body: JSON.stringify({ action: 'get_map', program: student.program, grade: student.grade })

      });

      const data = await resp.json();

      

      if(data.result === 'success' && data.courses.length > 0) {

        loadedCourses = data.courses;

        renderCourses(student, loadedCourses);

      } else {

        container.innerHTML = `

          <div class="student-badge">${student.name} (${student.grade})</div>

          <p style="color:#ef4444; font-size:14px; text-align:center;">매핑된 학습 코스가 없습니다.<br>지원서의 '학년' 형식과 Custom Programs 시트의 'Grade' 형식이 일치하는지 확인해 주세요.</p>`;

        document.getElementById('submitBtn').style.display = 'none';

      }

    } catch(e) { alert("코스 맵핑 로드 오류"); }

  }


  // [3단계] 체크박스 리스트 렌더링 (기본적으로 모두 체크됨)

  function renderCourses(student, courses) {

    let html = `<div class="student-badge">${student.name} (${student.program} / ${student.grade})</div>`;

    html += `<p style="font-size:13px; color:#475569; margin-bottom:15px;">기본 배정된 과목입니다. 수강하지 않을 과목은 체크를 해제해 주세요.</p>`;


    // Details(예: English / W&G) 기준으로 그룹화

    const grouped = courses.reduce((acc, obj) => {

      acc[obj.details] = acc[obj.details] || [];

      acc[obj.details].push(obj);

      return acc;

    }, {});


    for (const [detailName, items] of Object.entries(grouped)) {

      html += `<div class="course-card"><div class="course-header">📘 ${detailName}</div>`;

      items.forEach((item, idx) => {

        // value에 객체 정보를 담아두어 나중에 서버로 보낼 때 사용

        const itemData = encodeURIComponent(JSON.stringify(item));

        html += `

          <div class="checkbox-item">

            <input type="checkbox" id="chk_${detailName}_${idx}" value="${itemData}" checked>

            <label for="chk_${detailName}_${idx}">

              <span style="font-weight:bold;">${item.folder}</span>

              <span class="file-name">교재: ${item.file}</span>

            </label>

          </div>

        `;

      });

      html += `</div>`;

    }

    document.getElementById('studentContainer').innerHTML = html;

  }


  // [4단계] 선택된 코스 서버(Classroom Roster)로 전송

  async function submitCourses() {

    const checkboxes = document.querySelectorAll('input[type="checkbox"]:checked');

    if(checkboxes.length === 0) return alert("최소 1개 이상의 코스를 선택해야 합니다.");


    if(!confirm(`총 ${checkboxes.length}개의 세부 코스를 확정하시겠습니까?`)) return;


    const btn = document.getElementById('submitBtn');

    btn.disabled = true; btn.innerText = "제출 중...";


    const selectedCourses = Array.from(checkboxes).map(chk => JSON.parse(decodeURIComponent(chk.value)));


    try {

      const resp = await fetch(SCRIPT_URL, {

        method: 'POST',

        headers: { 'Content-Type': 'text/plain;charset=utf-8' },

        body: JSON.stringify({ 

          action: 'submit_courses', 

          childName: currentStudent.name,

          program: currentStudent.program,

          grade: currentStudent.grade,

          selectedCourses: selectedCourses

        })

      });

      const data = await resp.json();

      

      if(data.result === 'success') {

        alert("세부 코스 신청이 완료되었습니다!\nClassroom Roster에 정상 배정되었습니다.");

        location.reload();

      } else alert("오류: " + data.message);

    } catch(e) { alert("제출 중 오류가 발생했습니다."); }

    finally { btn.disabled = false; btn.innerText = "선택 코스 최종 제출하기"; }

  }

</script>
